import { createClient } from '@/lib/supabase/server'

export async function POST(request: Request) {
  const apiKey = process.env.ANTHROPIC_API_KEY
  if (!apiKey) return Response.json({ error: 'Claude API key not configured' }, { status: 500 })

  const response = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'x-api-key': apiKey,
      'anthropic-version': '2023-06-01',
    },
    body: JSON.stringify({ model: 'claude-sonnet-4-6', max_tokens: 2048, messages: [...] }),
  })
  if (!response.ok) {
    const errBody = await response.json().catch(() => ({}))
    return Response.json({ error: 'Claude API call failed', status: response.status, detail: errBody }, { status: 502 })
  }
  const data = await response.json()
  const text = data.content?.[0]?.text ?? ''
  const clean = text.replace(/^```(?:json)?\s*/i, '').replace(/\s*```\s*$/i, '').trim()
  return Response.json(JSON.parse(clean))
}
